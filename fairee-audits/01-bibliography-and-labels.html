Reproduce

cd "FaiREE_Missing_Data/Fairness with missing/Manuscript"

# Citation consistency
grep -hoP '@\w+\{\K[a-zA-Z0-9_]+' reflist_drafts.bib | sort -u > /tmp/bib_keys.txt
cd _drafts
grep -hoP '\\cite[tp]?\{\K[^}]+' *.tex | tr ',' '\n' | tr -d ' ' | sort -u > /tmp/cite_keys.txt
comm -23 /tmp/cite_keys.txt /tmp/bib_keys.txt    # cited-but-undefined (should be empty)
comm -13 /tmp/cite_keys.txt /tmp/bib_keys.txt    # defined-but-uncited (4 entries)

# Label consistency
grep -hoP '\\label\{\K[^}]+' *.tex | sort -u > /tmp/labels.txt
grep -hoP '\\(ref|eqref|cref|autoref)\{\K[^}]+' *.tex | sort -u > /tmp/refs.txt
comm -23 /tmp/refs.txt /tmp/labels.txt           # 5 residual dangling refs
