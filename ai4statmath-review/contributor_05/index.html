\begin{problemstatement}
Consider a sample of $n$ iid observations $O_i = (A_iY_i, A_i, X_i)$, $1 \leq i \leq n$, where $X \in [0,1]^d$, $A \in \{0, 1\}$ and $Y \in \{0, 1\}$. Suppose the estimand is $\psi = \int b(x) dP(x)$. Given estimators $\widehat{r}(x)$ and $\widehat{b}(x)$, bounded away from 0 and 1, consider the following ``structure-agnostic'' class of densities (see \url{https://arxiv.org/pdf/2305.04116}): \begin{align*} \mathcal{P}(\epsilon_n) = \left\{p: \int \{\widehat{r}(x) - r(x)\}^2 dP(x) \leq \epsilon^2_n , \quad \int \{\widehat{b}(x) - b(x)\}^2 dP(x) \leq \epsilon^2_n \right\}. \end{align*} Prove that \begin{align*} \inf_{\psi_n} \sup_{p \in \mathcal{P}(\epsilon_n)} E_p|\psi_n - \psi_p| \gtrsim \epsilon_n^2. \end{align*}
\end{problemstatement}
