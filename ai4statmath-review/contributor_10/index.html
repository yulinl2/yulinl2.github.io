By Cauchy--Schwarz, \begin{equation}\label{eq:smooth} \langle \nabla g(x) - \nabla g(y),\; x - y \rangle \;\le\; \lVert \nabla g(x) - \nabla g(y) \rVert \cdot \lVert x - y \rVert \;\le\; L \lVert x - y \rVert^{2}. \end{equation} Since $h$ is convex and differentiable, $\nabla h$ is a monotone operator: \begin{equation}\label{eq:mono} \langle \nabla h(x) - \nabla h(y),\; x - y \rangle \;\ge\; 0 \quad \text{for all } x, y \in \mathbb{R}^{d}. \end{equation} Writing $\nabla f = \nabla g - \nabla h$ and subtracting \eqref{eq:mono} from \eqref{eq:smooth} yields \eqref{eq:onesided}. \hfill$\square$ \medskip \noindent \textbf{Direction 2.} \emph{One-sided gradient inequality $\Rightarrow$ representation.} Suppose $f$ satisfies \eqref{eq:onesided}. Fix any $x_{0} \in \mathbb{R}^{d}$ and define \begin{align} g(x) &:= f(x_{0}) + \langle \nabla f(x_{0}),\, x - x_{0} \rangle + \tfrac{L}{2}\lVert x - x_{0} \rVert^{2}, \label{eq:g}\\ h(x) &:= g(x) - f(x). \label{eq:h} \end{align} We verify the three required properties. \smallskip \noindent (i)~\emph{$g$ is $L$-smooth.} By direct computation $\nabla g(x) = \nabla f(x_{0}) + L(x - x_{0})$, so $\lVert \nabla g(x) - \nabla g(y)\rVert = L\lVert x - y\rVert$. \smallskip \noindent (ii)~\emph{$h$ is continuously differentiable.} Since $f$ and $g$ are both continuously differentiable, so is $h = g - f$. \smallskip \noindent (iii)~\emph{$h$ is convex.} By the mean-value form of Taylor's theorem, there exists $t \in [0,1]$ such that \begin{equation}\label{eq:taylor} h(x) = h(y) + \langle \nabla h(y),\, x - y \rangle + \bigl\langle \nabla h\bigl(y + t(x-y)\bigr) - \nabla h(y),\, x - y \bigr\rangle. \end{equation} It suffices to show the last inner product is non-negative. Substituting $\nabla h = \nabla g - \nabla f$ and using the explicit form of $\nabla g$: \begin{align*} \bigl\langle \nabla h(y+t(x-y)) - \nabla h(y),\; x-y \bigr\rangle &= L\lVert x-y\rVert^{2} - \tfrac{1}{t}\bigl\langle \nabla f\bigl(y+t(x-y)\bigr) - \nabla f(y),\; t(x-y) \bigr\rangle \\ &\ge L\lVert x-y\rVert^{2} - Lt\lVert x-y\rVert^{2} \\ &\qquad \text{[by \eqref{eq:onesided} applied to the pair $\bigl(y+t(x-y),\,y\bigr)$]} \\ &\ge 0. \qquad \text{[since } 0 \le t \le 1\text{]} \end{align*} Hence $h(x) \ge h(y) + \langle \nabla h(y), x-y\rangle$ for all $x, y$, establishing convexity of $h$. \hfill$\square$ \medskip \noindent \textbf{Conclusion.} The largest class is the set of continuously differentiable functions satisfying \eqref{eq:onesided} --- equivalently, the \emph{$L$-weakly convex differentiable functions}.

\begin{problemstatement}
Let $f \colon \mathbb{R}^{d} \to \mathbb{R}$ be a continuously differentiable
function that admits the representation
\[
 f(x) = g(x) - h(x) \quad \text{for all } x \in \mathbb{R}^{d},
\]
where $g \colon \mathbb{R}^{d} \to \mathbb{R}$ has $L$-Lipschitz gradient
(i.e.\ $\lVert \nabla g(x) - \nabla g(y) \rVert \le L \lVert x - y \rVert$
for all $x, y$), and $h \colon \mathbb{R}^{d} \to \mathbb{R}$ is a
differentiable convex function.
Throughout, $f$ is \emph{not} assumed to be twice differentiable.

Identify the largest class of continuously differentiable functions $f$ that
always admits such a representation.
\end{problemstatement}

% ---------------------------------------------------------------
\begin{answerkey}
\textbf{Claim.}
A continuously differentiable function $f \colon \mathbb{R}^{d} \to \mathbb{R}$
admits a representation $f = g - h$ with $g$ having $L$-Lipschitz gradient and
$h$ differentiable convex \emph{if and only if} $f$ is $L$-weakly convex,
i.e.\ its gradient satisfies the one-sided inequality
\begin{equation}\label{eq:onesided}
 \langle \nabla f(x) - \nabla f(y),\; x - y \rangle
 \;\le\; L \lVert x - y \rVert^{2}
 \quad \text{for all } x, y \in \mathbb{R}^{d}.
\end{equation}

\medskip
\noindent
\textbf{Direction 1.}
\emph{Representation $\Rightarrow$ one-sided gradient inequality.}

Suppose $f = g - h$ with $\nabla g$ being $L$-Lipschitz and $h$ differentiable
convex.
By Cauchy--Schwarz,
\begin{equation}\label{eq:smooth}
 \langle \nabla g(x) - \nabla g(y),\; x - y \rangle
 \;\le\; \lVert \nabla g(x) - \nabla g(y) \rVert \cdot \lVert x - y \rVert
 \;\le\; L \lVert x - y \rVert^{2}.
\end{equation}
Since $h$ is convex and differentiable, $\nabla h$ is a monotone operator:
\begin{equation}\label{eq:mono}
 \langle \nabla h(x) - \nabla h(y),\; x - y \rangle \;\ge\; 0
 \quad \text{for all } x, y \in \mathbb{R}^{d}.
\end{equation}
Writing $\nabla f = \nabla g - \nabla h$ and subtracting
\eqref{eq:mono} from \eqref{eq:smooth} yields \eqref{eq:onesided}.
\hfill$\square$

\medskip
\noindent
\textbf{Direction 2.}
\emph{One-sided gradient inequality $\Rightarrow$ representation.}

Suppose $f$ satisfies \eqref{eq:onesided}.
Fix any $x_{0} \in \mathbb{R}^{d}$ and define
\begin{align}
 g(x) &:= f(x_{0}) + \langle \nabla f(x_{0}),\, x - x_{0} \rangle
 + \tfrac{L}{2}\lVert x - x_{0} \rVert^{2}, \label{eq:g}\\
 h(x) &:= g(x) - f(x). \label{eq:h}
\end{align}
We verify the three required properties.

\smallskip
\noindent
(i)~\emph{$g$ is $L$-smooth.}
By direct computation $\nabla g(x) = \nabla f(x_{0}) + L(x - x_{0})$, so
$\lVert \nabla g(x) - \nabla g(y)\rVert = L\lVert x - y\rVert$.

\smallskip
\noindent
(ii)~\emph{$h$ is continuously differentiable.}
Since $f$ and $g$ are both continuously differentiable, so is $h = g - f$.

\smallskip
\noindent
(iii)~\emph{$h$ is convex.}
By the mean-value form of Taylor's theorem, there exists $t \in [0,1]$ such that
\begin{equation}\label{eq:taylor}
 h(x) = h(y) + \langle \nabla h(y),\, x - y \rangle
 + \bigl\langle \nabla h\bigl(y + t(x-y)\bigr) - \nabla h(y),\,
 x - y \bigr\rangle.
\end{equation}
It suffices to show the last inner product is non-negative.
Substituting $\nabla h = \nabla g - \nabla f$ and using the explicit form of
$\nabla g$:
\begin{align*}
 \bigl\langle \nabla h(y+t(x-y)) - \nabla h(y),\; x-y \bigr\rangle
 &= L\lVert x-y\rVert^{2}
 - \tfrac{1}{t}\bigl\langle
 \nabla f\bigl(y+t(x-y)\bigr) - \nabla f(y),\; t(x-y)
 \bigr\rangle \\
 &\ge L\lVert x-y\rVert^{2} - Lt\lVert x-y\rVert^{2} \\
 &\qquad \text{[by \eqref{eq:onesided} applied to the pair
 $\bigl(y+t(x-y),\,y\bigr)$]} \\
 &\ge 0.
 \qquad \text{[since } 0 \le t \le 1\text{]}
\end{align*}
Hence $h(x) \ge h(y) + \langle \nabla h(y), x-y\rangle$ for all $x, y$,
establishing convexity of $h$.
\hfill$\square$

\medskip
\noindent
\textbf{Conclusion.}
The largest class is the set of continuously differentiable functions satisfying
\eqref{eq:onesided} --- equivalently, the \emph{$L$-weakly convex differentiable
functions}.
\end{answerkey}

\begin{problemstatement}
Let $f \colon \mathbb{R}^{d} \to \mathbb{R}$ be a continuously differentiable
function that is bounded below by $f^{*} > -\infty$.
It is well known that for an $M$-smooth function, gradient descent with stepsize
$\alpha = 1/M$ guarantees sufficient descent, and the optimal stepsize can be
read off directly from the smoothness parameter $M$.

Now suppose $f$ admits the representation
\[
 f \;=\; g - h,
\]
where $g \colon \mathbb{R}^{d} \to \mathbb{R}$ has $L$-Lipschitz gradient and
$h \colon \mathbb{R}^{d} \to \mathbb{R}$ is a smooth convex function.
Throughout, $f$ is \emph{not} assumed to be twice differentiable.

What is the largest stepsize $\alpha$, up to universal constants, that guarantees
sufficient descent for gradient descent on $f$\,?
Prove that your claimed stepsize is valid and that no larger stepsize is valid
up to universal constants.
\end{problemstatement}

% ---------------------------------------------------------------
\begin{answerkey}

\paragraph{Step 1 --- The decomposition is not unique; $1/L$ is suboptimal.}
The representation $f = g - h$ is not unique: for any smooth convex $\phi$,
the pair $(g + \phi,\, h + \phi)$ is another valid decomposition with
potentially a smaller Lipschitz constant for the smooth part.
Hence $1/L$ is valid but generally suboptimal.

\paragraph{Step 2 --- Effective smoothness.}
Optimizing over all valid decompositions defines
\begin{equation}\label{eq:Mstar}
 M^{*}_{f}
 \;:=\;
 \inf_{h}\bigl\{\,L \;\big|\;
 (f + h)\text{ is }L\text{-smooth},\; h\text{ smooth convex}\bigr\}.
\end{equation}
The largest valid stepsize up to universal constants is
$\alpha^{*} \asymp 1/M^{*}_{f}$.

\paragraph{Step 3 --- Validity of $\alpha = 1/M^{*}_{f}$ (self-contained).}
Let $h^{*}$ achieve (or approximate) the infimum in \eqref{eq:Mstar} and set
$g^{*}:= f + h^{*}$, which is $M^{*}_{f}$-smooth.
The standard descent lemma applied to $g^{*}$ gives, for all
$\alpha \le 1/M^{*}_{f}$:
\[
 g^{*}\!\left(x - \alpha\nabla g^{*}(x)\right)
 \;\le\; g^{*}(x) - \tfrac{\alpha}{2}\lVert\nabla g^{*}(x)\rVert^{2}.
\]
Since $\nabla g^{*} = \nabla f + \nabla h^{*}$ and $h^{*}$ is convex,
monotonicity of $\nabla h^{*}$ gives
$h^{*}(x - \alpha\nabla f(x))
\ge h^{*}(x) - \alpha\langle\nabla h^{*}(x), \nabla f(x)\rangle$.
Using the identity
$\lVert\nabla g^{*}(x)\rVert^{2}
= \lVert\nabla f(x)\rVert^{2}
+ 2\langle\nabla f(x), \nabla h^{*}(x)\rangle
+ \lVert\nabla h^{*}(x)\rVert^{2}$
and $f = g^{*} - h^{*}$, combining these bounds yields
\begin{equation}\label{eq:descent}
 f\!\left(x - \alpha\nabla f(x)\right)
 \;\le\; f(x) - \tfrac{\alpha}{2}\lVert\nabla f(x)\rVert^{2}
 \quad\text{for all } \alpha \le \tfrac{1}{M^{*}_{f}}.
\end{equation}
Since $f \ge f^{*}$, telescoping \eqref{eq:descent} over $k$ iterations gives
\begin{equation}\label{eq:rate-known}
 \operatorname{Avg}\!\left(\lVert\nabla f(x^{k})\rVert^{2}\right)
 \;\le\;
 \frac{2\bigl(f(x^{0}) - f^{*}\bigr)}{\alpha(k+1)}.
 \tag{5a}
\end{equation}

\paragraph{Step 4 --- $M^{*}_{f}$ is unknown; use backtracking.}
Since $M^{*}_{f}$ is not computable without the optimal decomposition,
use backtracking line search with parameter $\beta \in (0,1)$: at iteration $k$
choose the smallest non-negative integer $i_{k}$ such that
$t^{k}:= \beta^{i_{k}}$ satisfies
\begin{equation}\label{eq:armijo}
 f\!\left(x^{k} - t^{k}\nabla f(x^{k})\right)
 \;\le\; f(x^{k}) - \tfrac{t^{k}}{2}\lVert\nabla f(x^{k})\rVert^{2}.
 \tag{39}
\end{equation}
By \eqref{eq:descent}, \eqref{eq:armijo} holds whenever
$t^{k} \le 1/M^{*}_{f}$, so backtracking terminates with
$t^{k} \ge \min\!\left\{1,\, \beta/M^{*}_{f}\right\}$.
Applying \eqref{eq:armijo} repeatedly and using $f \ge f^{*}$:
\begin{equation}\label{eq:rate-bt}
 \operatorname{Avg}\!\left(\lVert\nabla f(x^{k})\rVert^{2}\right)
 \;\le\;
 \frac{2\max\!\left\{1,\, M^{*}_{f}/\beta^{2}\right\}
 \bigl(f(x^{0}) - f^{*}\bigr)}{\beta^{2}(k+1)}.
 \tag{5b}
\end{equation}

\paragraph{Step 5 --- Tightness up to universal constants.}
For any $\alpha > 2/M^{*}_{f}$, consider
$f(x) = \tfrac{M^{*}_{f}}{2}\lVert x\rVert^{2}$, which is bounded below by
$f^{*} = 0$ and achieves the infimum in \eqref{eq:Mstar}.
Gradient descent gives $x^{k+1} = (1 - \alpha M^{*}_{f})\,x^{k}$; since
$\alpha M^{*}_{f} > 2$, we have $|1 - \alpha M^{*}_{f}| > 1$, so
$\lVert x^{k}\rVert \to \infty$.
Hence no stepsize $\alpha > 2/M^{*}_{f}$ is universally valid, giving
tightness up to the universal constant $2$:
\[
 \alpha^{*} \;\asymp\; \frac{1}{M^{*}_{f}}.
\]
\hfill$\square$

\noindent\textbf{Remark.}
Note that for $\alpha \in (1/M^{*}_{f},\, 2/M^{*}_{f})$, the iterates on this
quadratic example converge but the sufficient descent condition
\eqref{eq:descent} fails.
The tightness proof therefore requires $\alpha > 2/M^{*}_{f}$ for a divergence
argument; claiming divergence at $\alpha > 1/M^{*}_{f}$ is incorrect.
\end{answerkey}

\begin{problemstatement}
Let $\{X_{1,t}\}_{t \ge 1}$ and $\{X_{2,t}\}_{t \ge 1}$ be independent
sequences of i.i.d.\ $1$-sub-Gaussian random variables with means
$\mu_{1} = \mu$ and $\mu_{2} = \mu - \delta$, where $\delta > 0$ satisfies
$\delta \cdot \sqrt{T/(2\log T)} = o(1)$ (so that both arms are pulled
infinitely often).
Consider the fixed-horizon UCB algorithm that at each time
$t \in \{1, \ldots, T\}$ selects the arm
\[
 A_{t}
 = \operatorname*{arg\,max}_{i \in \{1,2\}}
 \left\{
 \hat{\mu}_{i}(t-1) + \sqrt{\frac{2\log T}{N_{i}(t-1)}}
 \right\},
\]
where $N_{i}(t) = \sum_{s=1}^{t}\mathbf{1}[A_{s}=i]$ and
$\hat{\mu}_{i}(t) = N_{i}(t)^{-1}\sum_{s=1}^{t}X_{i,s}\mathbf{1}[A_{s}=i]$.

Give a complete characterization of $N_{2}(T)$ as a function of $\delta$
and $T$, \emph{uniformly} over all $\delta$ satisfying the above condition.
A characterization valid only for fixed $\delta > 0$ as $T \to \infty$
receives no credit.

\begin{enumerate}
 \item Derive a single expression $n^{\omega} = n^{\omega}(\delta, T)$
 such that the upper bound
 $N_{2}(T) \lesssim n^{\omega}(1 + o(1))$ in probability using only
 the UCB rule and a concentration argument.
 Do not invoke general bandit theorems; the proof must be
 self-contained for $K = 2$.

 \item Prove the matching lower bound
 $N_{2}(T) \gtrsim n^{\omega}(1 - o(1))$ in probability.

 \item Verify that your expression $n^{\omega}$ is consistent with
 Problem~3 as $\delta \to 0$.
\end{enumerate}

\begin{answerkey}
Define the good event
\[
 \mathcal{E}_{T}
 = \Bigl\{
 \lvert\hat{\mu}_{i,t} - \mu_{i}\rvert

 g_{T} = \sqrt{7\log\log T + 3\log 2}.
\]
By the sub-Gaussian LIL, $\mathbb{P}(\mathcal{E}_{T}) \ge 1 - 6/\log T \to 1$.
All arguments below are on $\mathcal{E}_{T}$.

\medskip
\noindent\textbf{The characteristic equation and $n^{\omega}$.}

Define $n^{\omega} = n^{\omega}(\delta, T)$ as the unique solution in
$(0, T)$ to
\begin{equation}\label{eq:char}
 \frac{1}{\Bigl(\sqrt{T/n^{\omega}}
 + \sqrt{T\delta^{2}/(2\log T)}\Bigr)^{2}}
 +
 \frac{n^{\omega}}{T}
 \;=\; 1.
\end{equation}
Equation~\eqref{eq:char} has a unique solution because the left-hand side
is strictly increasing in $n^{\omega}$ on $(0, T)$, equals $0$ at
$n^{\omega} = 0$, and exceeds $1$ at $n^{\omega} = T$ (for $K > 1$).

\medskip
\noindent\textbf{Part~(1): Upper bound $N_{2}(T) \lesssim n^{\omega}(1+o(1))$.}

Let $T_{2}$ be the last time arm~2 is pulled.
By the UCB rule at time $T_{2}$:
\begin{equation}\label{eq:ucb-ub}
 \hat{\mu}_{2,T_{2}} + \sqrt{\frac{2\log T}{N_{2,T_{2}}}}
 \;\ge\;
 \hat{\mu}_{1,T_{2}} + \sqrt{\frac{2\log T}{N_{1,T_{2}}}}.
\end{equation}
On $\mathcal{E}_{T}$, apply
$\hat{\mu}_{1,T_{2}} \ge \mu - g_{T}/\sqrt{N_{1}(T)}$ and
$\hat{\mu}_{2,T_{2}} \le \mu - \delta + g_{T}/\sqrt{N_{2}(T)}$
to \eqref{eq:ucb-ub}, use $N_{2,T_{2}} = N_{2}(T) - 1$ and
$N_{1,T_{2}} \le N_{1}(T) = T - N_{2}(T)$, to obtain
\begin{equation}\label{eq:ub-sandwich}
 \sqrt{\frac{2\log T}{N_{2}(T)}}
 + \frac{2g_{T}}{\sqrt{N_{2}(T)}}
 \;\ge\;
 \sqrt{\frac{2\log T}{T - N_{2}(T)}} + \delta.
\end{equation}
Dividing both sides by $\sqrt{2\log T/T}$ and setting
$\rho = N_{2}(T)/T$:
\begin{equation}\label{eq:rho-ub}
 \frac{1}{\sqrt{\rho}}
 \;\ge\;
 \frac{1}{\sqrt{1-\rho}}
 + \frac{\delta}{\sqrt{2\log T/T}}
 - \frac{2g_{T}}{\sqrt{2\log T}}
 \;=\;
 \frac{1}{\sqrt{1-\rho}} + \kappa - o(1),
\end{equation}
where $\kappa:= \delta\sqrt{T/(2\log T)} = o(1)$ by assumption.

can be written $1/\sqrt{\rho^{\omega}} = 1/\sqrt{1-\rho^{\omega}} + \kappa$

 \hat{\mu}_{2,T_{1}} + \sqrt{\frac{2\log T}{N_{2,T_{1}}}}.
\end{equation}
